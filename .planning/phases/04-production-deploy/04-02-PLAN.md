  - Define the production base URL: `https://3-d-tl-design.vercel.app`
  - Create `public/qr/` directory if it does not exist
  - For each project in CLIENT_PROJECTS:
    - Generate URL: `${BASE_URL}/project/${project.code}`
    - Generate PNG file: `public/qr/${project.code}.png` (width: 512px, margin: 2, dark color: #0d0d0f for brand alignment, error correction level: H for maximum reliability)
    - Generate SVG string and write to: `public/qr/${project.code}.svg`
    - Log: `Generated QR for ${project.clientName} — ${project.projectName}: ${url}`
  - Log summary: total QR codes generated, output directory

  Implementation using qrcode library:

  ```typescript
  import QRCode from 'qrcode'
  import { mkdirSync, writeFileSync } from 'fs'
  import { join } from 'path'

  // Import project data
  // Note: tsx can resolve .ts imports
  import { CLIENT_PROJECTS } from '../src/data/projects.ts'

  const BASE_URL = 'https://3-d-tl-design.vercel.app'
  const OUTPUT_DIR = join(process.cwd(), 'public', 'qr')

  const QR_OPTIONS = {
    width: 512,
    margin: 2,
    color: {
      dark: '#0d0d0f',   // matches brand primary background for subtle branding
      light: '#ffffff',
    },
    errorCorrectionLevel: 'H' as const,
  }

  async function main() {
    mkdirSync(OUTPUT_DIR, { recursive: true })

    for (const project of CLIENT_PROJECTS) {
      const url = `${BASE_URL}/project/${project.code}`

      // PNG
      await QRCode.toFile(
        join(OUTPUT_DIR, `${project.code}.png`),
        url,
        QR_OPTIONS,
      )

      // SVG
      const svg = await QRCode.toString(url, {
        ...QR_OPTIONS,
        type: 'svg',
      })
      writeFileSync(join(OUTPUT_DIR, `${project.code}.svg`), svg)

      console.log(`Generated QR for ${project.clientName} - ${project.projectName}: ${url}`)
    }

    console.log(`\nDone: ${CLIENT_PROJECTS.length} QR codes generated in ${OUTPUT_DIR}`)
  }

  main().catch(console.error)
  ```
